Download the ptouch-print git repository and compile it.

sudo apt install -y autogen autoconf gettext autopoint gcc libgd-dev libusb-dev libusb-1.0-0-dev make autofs udev dialog
cd ptouch-print
./autogen.sh
./configure
make
